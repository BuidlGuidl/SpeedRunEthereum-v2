// ── Camera ──────────────────────────────────────────────
// Fixed transform-origin at the right edge / 90% height of the shot:
// zooming in anchors the panel to the bottom-right of the viewport
// (panel header crops off-screen — same convention as AITutorDemo's chat zoom).
const ZOOM_SCALE = 1.5;
const ZOOM_EASE = 45;

function getCameraScale(f: number): number {
  const zoomIn = interpolate(f, [T.ZOOM_IN, T.ZOOM_IN + ZOOM_EASE], [0, 1], {
    ...EC,
    easing: Easing.inOut(Easing.cubic),
  });
  const zoomOut = interpolate(f, [T.ZOOM_OUT, T.ZOOM_OUT + 50], [1, 0], {
    ...EC,
    easing: Easing.inOut(Easing.cubic),
  });
  return 1 + (ZOOM_SCALE - 1) * zoomIn * zoomOut;
}

// ── Full-screen chapter card → shrinks to a compact top badge (AITutorDemo pattern) ──
const ChapterCard: React.FC<{
  num: string;
  title: string;
  sub: string;
  start: number;
  end: number;
  persistUntil: number;
}> = ({ num, title, sub, start, end, persistUntil }) => {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();
  const dur = end - start;
  const holdEnd = start + Math.floor(dur * 0.55);
  const shrinkEnd = start + Math.floor(dur * 0.72);
  const transEnd = start + Math.floor(dur * 0.85);

  const fadeIn = interpolate(frame, [start, start + 15], [0, 1], EC);
  const s = spring({ frame: Math.max(0, frame - start), fps, config: { damping: 18, mass: 0.8 } });
  const fadeOut = interpolate(frame, [persistUntil - 15, persistUntil], [1, 0], EC);
  if (frame < start || frame >= persistUntil) return null;

  const isPersisting = frame >= end;
  const shrinkProg = isPersisting ? 1 : frame <= holdEnd ? 0 : interpolate(frame, [holdEnd, shrinkEnd], [0, 1], EC);
  const slideRaw = isPersisting ? 1 : frame <= shrinkEnd ? 0 : interpolate(frame, [shrinkEnd, transEnd], [0, 1], EC);
  const slideProg = Easing.out(Easing.cubic)(slideRaw);

  const yPos = interpolate(slideProg, [0, 1], [50, 4.5]);
  const scale = interpolate(shrinkProg, [0, 1], [1, 0.52]);
  const subOp = interpolate(shrinkProg, [0, 0.35], [1, 0], EC);
  const subMargin = interpolate(shrinkProg, [0, 0.35], [20, 0], EC);
  const subMaxH = interpolate(shrinkProg, [0, 0.35], [120, 0], EC);
  const bgOp = interpolate(shrinkProg, [0, 0.5], [0.92, 0.85], EC);
  const padV = interpolate(shrinkProg, [0, 1], [60, 14]);
  const padH = interpolate(shrinkProg, [0, 1], [120, 44]);
  const numMb = interpolate(shrinkProg, [0, 1], [16, 6]);
  const op = isPersisting ? fadeOut : fadeIn;

  return (
    <AbsoluteFill style={{ justifyContent: "flex-start", alignItems: "center", zIndex: 50, opacity: op, pointerEvents: "none" }}>
      <div
        style={{
          position: "absolute",
          top: `${yPos}%`,
          left: "50%",
          transform: `translate(-50%, -50%) scale(${(isPersisting ? 1 : s) * scale})`,
          background: `rgba(10,10,18,${bgOp})`,
          backdropFilter: "blur(16px)",
          padding: `${padV}px ${padH}px`,
          borderRadius: 24,
          textAlign: "center",
          border: "1px solid rgba(103,221,222,0.25)",
          boxShadow: "0 32px 80px rgba(2,98,98,0.35)",
          display: "flex",
          flexDirection: "column" as const,
          alignItems: "center",
        }}
      >
        <div style={{ color: colors.accent, fontSize: 36, fontFamily: fonts.mono, fontWeight: 700, marginBottom: numMb, letterSpacing: 2, opacity: 0.85 }}>
          {num}
        </div>
        <h2 style={{ color: "#fff", fontSize: 76, fontFamily: fonts.heading, fontWeight: 800, margin: 0, letterSpacing: "-0.02em", whiteSpace: "nowrap" }}>
          {title}
        </h2>
        <div style={{ overflow: "hidden", maxHeight: subMaxH, marginTop: subMargin }}>
          <p style={{ color: "rgba(255,255,255,0.5)", fontSize: 33, fontFamily: fonts.body, margin: 0, opacity: subOp }}>{sub}</p>
        </div>
      </div>
    </AbsoluteFill>
  );
};
